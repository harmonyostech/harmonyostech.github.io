3. HarmonyOS AES decryption issues?

AES encryption and decryption are required in the app, using the Crypto Architecture Kit. The encryption and decryption code differences are not significant, mostly copied from official documentation. Encryption works, but decryption errors occur at the decoder.doFinal() position.

Refer to the following demo: 

import cryptoFramework from '@ohos.security.cryptoFramework';
import buffer from '@ohos.buffer';

function genIvParamsSpec() {
  let dataIv = convertStringToUint8Array('8NONwyJtHesysWpM')!; // new Uint8Array(arr);
  let ivBlob: cryptoFramework.DataBlob = { data: dataIv };
  let ivParamsSpec: cryptoFramework.IvParamsSpec = {
    algName: "IvParamsSpec",
    iv: ivBlob
  };
  return ivParamsSpec;
}

// Encrypt message
async function encryptMessagePromise(symKey: cryptoFramework.SymKey, plainText: cryptoFramework.DataBlob) {
  let cipher = cryptoFramework.createCipher('AES128|CBC|PKCS7');
  let iv = genIvParamsSpec();
  await cipher.init(cryptoFramework.CryptoMode.ENCRYPT_MODE, symKey, iv);
  let cipherData = await cipher.doFinal(plainText);
  return cipherData;
}

// Decrypt message
async function decryptMessagePromise(symKey: cryptoFramework.SymKey, cipherText: cryptoFramework.DataBlob) {
  let decoder = cryptoFramework.createCipher('AES128|CBC|PKCS7');
  let iv = genIvParamsSpec();
  await decoder.init(cryptoFramework.CryptoMode.DECRYPT_MODE, symKey, iv);
  let decryptData = await decoder.doFinal(cipherText);
  return decryptData;
}

async function genSymKeyByData(symKeyData: Uint8Array) {
  let symKeyBlob: cryptoFramework.DataBlob = { data: symKeyData };
  let aesGenerator = cryptoFramework.createSymKeyGenerator('AES128');
  let symKey = await aesGenerator.convertKey(symKeyBlob);
  console.info('convertKey success');
  return symKey;
}

async function aesCBC() {
  let keyData = convertStringToUint8Array('sgg45747ss223455')!;
  let symKey = await genSymKeyByData(keyData);
  let message = "This is a test,just";

    data: new Uint8Array(buffer.from(message, 'utf-8').buffer) 
  };
  let encryptText = await encryptMessagePromise(symKey, plainText);

1. HarmonyOS Bluetooth Key Module?

After setting the setCharacteristicChangeNotification in the Bluetooth key module, the characteristic value change cannot be monitored in the BLECharacteristicChange subscription event.
Steps:

  1. Call the setCharacteristicChangeNotification interface to set the notify property of the characteristic to true.
  2. Call the writeDescriptorValue interface to set the descriptor value of the characteristic to {0x01, 0x00}.
  3. Call the writeCharacteristicValue interface to successfully send 7 - byte data to the master device.
  4. The returned data cannot be monitored in the BLECharacteristicChange callback.

If the device itself supports data return after writing data, it is necessary to check whether the interface method of writeCharacteristicValue is inconsistent with the service of the device that will return data.
